<
PreLaunchStep
>

,
)

,

,

:
GuestProgramLauncherComponent
,
gameSource
:
GameSource
,
) {
for
(step

preLaunchSteps) {
if
(
!
step.appliesTo(container, appId, gameSource))

try

step.
run
(context, appId, container, guestProgramLauncherComponent, gameSource)
}
catch
(e

Exception

.tag(
"
PreLaunchSteps
"

class
.simpleName}
"

}
}
